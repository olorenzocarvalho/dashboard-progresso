// Função para obter as tarefas da ClickUp
async function getTasks() {
  try {
    const response = await axios.get(`https://api.clickup.com/api/v2/list/${listId}/task`, {
      headers: {
        'Authorization': apiKey,  // Usando o token da API
      },
    });

    return response.data.tasks;
  } catch (error) {
    console.error('Erro ao obter tarefas:', error);
    return [];
  }
}

// Função para calcular o progresso com base nas etiquetas (considerando os valores 1, 0.5 e 0)
function calculateProgress(tasks) {
  let total = tasks.length;
  let progressSum = 0;

  // Contando as tarefas por etiqueta e somando o valor correspondente
  tasks.forEach(task => {
    if (task.status.status === 'ok') {
      progressSum += 1;  // 100% para "ok"
    } else if (task.status.status === 'em andamento') {
      progressSum += 0.5;  // 50% para "em andamento"
    } else if (task.status.status === 'restante') {
      progressSum += 0;  // 0% para "restante"
    }
  });

  // Calcular o progresso médio em porcentagem
  const progress = (progressSum / total) * 100;
  return progress;
}

// Função para obter os dados e calcular o progresso
async function getProgress() {
  const tasks = await getTasks();
  const progress = calculateProgress(tasks);
  console.log('Progresso Atualizado: ', progress);
  return progress;
}

// Exemplo de uso da função
getProgress().then(progress => {
  console.log('Progresso calculado:', progress);
}).catch(error => {
